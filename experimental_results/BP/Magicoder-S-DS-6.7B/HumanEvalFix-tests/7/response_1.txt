def find_max_sum(numbers: List[int]) -> int:
    max_sum = float('-inf')
    for i in range(len(numbers)):
        for j in range(i, len(numbers)):
            max_sum = max(max_sum, sum(numbers[i:j+1]))
    return max_sum
