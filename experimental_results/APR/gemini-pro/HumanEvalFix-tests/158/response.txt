def find_max(words):
    max_unique = 0
    max_word = ""
    for word in words:
        unique = set(word)
        if len(unique) > max_unique:
            max_unique = len(unique)
            max_word = word
        elif len(unique) == max_unique:
            if word < max_word:
                max_word = word
    return max_word
