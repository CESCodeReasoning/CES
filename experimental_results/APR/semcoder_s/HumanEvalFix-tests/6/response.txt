from typing import List


def find_duplicates(numbers: List[int]) -> List[int]:
    counts = {}
    for num in numbers:
        if num in counts:
            counts[num] += 1
        else:
            counts[num] = 1

    duplicates = []
    for num, count in counts.items():
        if count > 1:
            duplicates.append(num)

    return duplicates
